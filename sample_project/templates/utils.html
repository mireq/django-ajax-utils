Misc

_.has (replacement for Object.prototype.hasOwnProperty):

var obj = {'property': true};

dumpVar(_.has(obj, 'property'));

_.isEqual

var a = {'prop': 'x'};
var b = {'prop': 'y'};
var aCopy = {'prop': 'x'};

dumpVar(_.isEqual(a, b));
dumpVar(_.isEqual(a, aCopy));

_.deepCopy

var a = {'prop': {}};
var b = _.deepCopy(a);

dumpVar(a === b); // a is not same object as b
dumpVar(a.prop === b.prop); // properties are not same

_.lightCopy

var a = {'prop': {}};
var b = _.lightCopy(a);

dumpVar(a === b); // a is not same object as b
dumpVar(a.prop === b.prop); // properties are same
